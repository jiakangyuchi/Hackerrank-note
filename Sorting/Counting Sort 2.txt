Version 1

function countingSort(arr, n) {
    var map = {};
    var res = [];
    for(let i = 0; i < n; i ++){
        var target = arr[i];
        map[target] = (map[target] || 0) + 1;
    }
    for(let i = 0; i < 100; i ++){
        while(map[i] > 0){
            res.push(i);
            map[i] --;
        }
    }
    return res;
}//计数排序，时间复杂度O(n + k)

Version 2

function countingSort(arr, n) {
    return arr.sort((x, y) => x - y);
}//冒泡排序，时间复杂度O(n2)
