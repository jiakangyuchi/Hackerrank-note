function main() {
    var S = readLine();
    var result = 0;
    for(let i = 0; i < S.length - 2; i += 3){
        var str = S.substring(i, i + 3);
        if(str[0] !== "S") result ++;
        if(str[1] !== "O") result ++;
        if(str[2] !== "S") result ++;
    }
    console.log(result);
}
